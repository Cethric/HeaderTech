51  void OpenGLWindowApi::ToggleFullscreen() noexcept
52  {
53  GLFWmonitor *monitor = glfwGetWindowMonitor(m_window);
54  if (monitor) {
55  glfwSetWindowMonitor(m_window, nullptr, 200, 200, 800, 600, 0);
56  } else {
57  int monitorCount = 0;
58  GLFWmonitor **monitors = glfwGetMonitors(&monitorCount);
59  GLFWmonitor *activeMonitor = nullptr;
60 
61  int wx, wy;
62  glfwGetWindowPos(m_window, &wx, &wy);
63 
64  for (int i = 0; i < monitorCount; ++i) {
65  GLFWmonitor *current = *(monitors + i);
66  int x, y, width, height;
67  glfwGetMonitorWorkarea(current, &x, &y, &width, &height);
68 
69  SPDLOG_INFO("Checking window: {} {}x{} {}x{}", glfwGetMonitorName(current), x, y, width, height);
70 
71  if (wx > x && wy > y && wx < (x + width) && wy < (y + height)) {
72  activeMonitor = current;
73  }
74  }
75 
76  if (activeMonitor) {
77  SPDLOG_INFO(
78  "Using window: {} {}",
79  glfwGetMonitorName(activeMonitor),
80  glfwGetWin32Monitor(activeMonitor)
81  );
82  const GLFWvidmode *mode = glfwGetVideoMode(activeMonitor);
83 
84  glfwSetWindowMonitor(m_window, activeMonitor, 0, 0, mode->width, mode->height, mode->refreshRate);
85  }
86  }
87  }
