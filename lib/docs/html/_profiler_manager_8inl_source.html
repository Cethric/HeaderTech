97 
98  if (flags == Types::ScopedProfilerFlags_Recursive && parent_hash == hash) {
99  auto &current = m_cpuProfiles.at(hash);
100  ++current.call_count;

102  current.max_recursion_count = current.max_recursion_count > current.recursion_count
103  ? current.max_recursion_count
104  : current.recursion_count;
105  return;
106  }
107 
108  m_cpuProfiles.insert_or_assign(
109  hash,
110  {
111  .name=name,
112  .function=function,
113  .line=line,
114  .file=file,
115  .flags=flags,
116  .call_count=1,
117  .recursion_count=(flags == Types::ScopedProfilerFlags_Recursive ? 1u : 0u),
118  .max_recursion_count=(flags == Types::ScopedProfilerFlags_Recursive ? 1u : 0u),
119  .start=glfwGetTime(),
120  .end=0,
121  .delta=0,
122  .total_delta=0,
123  .parent=parent_hash,

125  }
126  );

128  m_currentCpuProfile.push_front(hash);

131  void ProfilerManager::EndCpuProfile() noexcept
132  {
133  if (m_currentCpuProfile.empty()) {
134  return;
135  }
136  auto current_hash = m_currentCpuProfile.front();
137  auto &current = m_cpuProfiles.at(current_hash);
138  if (current.recursion_count > 0) {
139  --current.recursion_count;
140  } else {
141  auto end_time = glfwGetTime();
142 
143  double delta;
144  if (current.call_count > 1 && current.max_recursion_count == 0) {
145  delta = end_time - current.end;
146  } else {
147  delta = end_time - current.start;
148  }
149  current.delta += delta;
150 
151  if (current.parent > 0) {
152  auto &parent = m_cpuProfiles.at(current.parent);
153  parent.total_delta += delta;
154  }
155 
156  m_currentCpuProfile.pop_front();
157  if (current.flags == Types::ScopedProfilerFlags_Recursive) {
158  m_currentCpuProfile.pop_front();
159  }
160  }
161  }
