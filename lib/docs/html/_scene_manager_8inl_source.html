39  inline void SceneManager::PushScene()
40  {
41  if (m_active->m_children.empty()) {
42  SceneGraph *active = m_active;
43  while (active) {
44  if (active->m_parent == nullptr) {
45  break;
46  }
47  auto &children = active->m_parent->m_children;
48  auto child = std::find(std::cbegin(children), std::cend(children), active);
49  if (child == std::cend(children)) {
50  SPDLOG_LOGGER_CRITICAL(m_log, "Unable to find the active scene in the parent's scenes");
51  break;
52  }
53  auto distance = std::distance(std::cbegin(children), child) + 1;
54  if (distance >= children.size()) {
55  active = active->m_parent;
56  continue;
57  }
58  PushNextScene(children.at(distance));
59  return;
60  }
61  SPDLOG_LOGGER_WARN(m_log, "No more scenes to push");
62  return;
63  }
64  PushNextScene(m_active->m_children.front());
65  }
